```ext2_ls.c
#include <stdio.h>
#include <stdint.h>
#include <fcntl.h>
#include <unistd.h>
#include <stdlib.h>
#include <string.h>

/*
 * ext2 のオンディスク構造を「一切省略せず」定義。
 * #pragma pack でアラインメントによる隙間を排除し、
 * オフセットがディスク上の配置と一致するようにする。
 */
#pragma pack(push, 1)

/* ---- スーパーブロック (1024バイト、オフセット1024に存在) ---- */
struct ext2_super_block {
    uint32_t s_inodes_count;
    uint32_t s_blocks_count;
    uint32_t s_r_blocks_count;
    uint32_t s_free_blocks_count;
    uint32_t s_free_inodes_count;
    uint32_t s_first_data_block;
    uint32_t s_log_block_size;
    uint32_t s_log_frag_size;
    uint32_t s_blocks_per_group;
    uint32_t s_frags_per_group;
    uint32_t s_inodes_per_group;
    uint32_t s_mtime;
    uint32_t s_wtime;
    uint16_t s_mnt_count;
    uint16_t s_max_mnt_count;
    uint16_t s_magic;
    uint16_t s_state;
    uint16_t s_errors;
    uint16_t s_minor_rev_level;
    uint32_t s_lastcheck;
    uint32_t s_checkinterval;
    uint32_t s_creator_os;
    uint32_t s_rev_level;
    uint16_t s_def_resuid;
    uint16_t s_def_resgid;
    uint32_t s_first_ino;
    uint16_t s_inode_size;
    uint16_t s_block_group_nr;
    uint32_t s_feature_compat;
    uint32_t s_feature_incompat;
    uint32_t s_feature_ro_compat;
    uint8_t  s_uuid[16];
    char     s_volume_name[16];
    char     s_last_mounted[64];
    uint32_t s_algorithm_usage_bitmap;
    uint8_t  s_prealloc_blocks;
    uint8_t  s_prealloc_dir_blocks;
    uint16_t s_padding1;
    uint8_t  s_journal_uuid[16];
    uint32_t s_journal_inum;
    uint32_t s_journal_dev;
    uint32_t s_last_orphan;
    uint32_t s_hash_seed[4];
    uint8_t  s_def_hash_version;
    uint8_t  s_reserved_char_pad;
    uint16_t s_reserved_word_pad;
    uint32_t s_default_mount_opts;
    uint32_t s_first_meta_bg;
    uint32_t s_reserved[190];
};

/* ---- ブロックグループデスクリプタ (32バイト) ---- */
struct ext2_group_desc {
    uint32_t bg_block_bitmap;
    uint32_t bg_inode_bitmap;
    uint32_t bg_inode_table;
    uint16_t bg_free_blocks_count;
    uint16_t bg_free_inodes_count;
    uint16_t bg_used_dirs_count;
    uint16_t bg_pad;
    uint32_t bg_reserved[3];
};

/* ---- inode (128バイト クラシック構成) ---- */
struct ext2_inode {
    uint16_t i_mode;
    uint16_t i_uid;
    uint32_t i_size;
    uint32_t i_atime;
    uint32_t i_ctime;
    uint32_t i_mtime;
    uint32_t i_dtime;
    uint16_t i_gid;
    uint16_t i_links_count;
    uint32_t i_blocks;
    uint32_t i_flags;
    uint32_t i_osd1;
    uint32_t i_block[15];
    uint32_t i_generation;
    uint32_t i_file_acl;
    uint32_t i_dir_acl;
    uint32_t i_faddr;
    uint8_t  i_frag;
    uint8_t  i_fsize;
    uint16_t i_pad1;
    uint16_t i_uid_high;
    uint16_t i_gid_high;
    uint32_t i_reserved2;
};

/* ---- ディレクトリエントリ (可変長) ---- */
struct ext2_dir_entry {
    uint32_t inode;
    uint16_t rec_len;
    uint8_t  name_len;
    uint8_t  file_type;
    uint8_t  name[];   /* name_len バイトのファイル名が続く */
};

#pragma pack(pop)

/*
[ブート]
[スーパーブロック]
[GDT][GDT][GDT]...   ← ここ！
[ブロックビットマップ]
[inodeビットマップ]
[inodeテーブル]
[データブロック]
*/

int main(int argc, char *argv[])
{
    // ext2.imgを指定していない
    if (argc != 2) {
        fprintf(stderr, "usage: %s ext2.img\n", argv[0]);
        return 1;
    }

    // ext2.imgを開けない
    int fd = open(argv[1], O_RDONLY);
    if (fd < 0) { perror("open"); return 1; }

    /* 1) スーパーブロック読み込み (常に1024 バイト目から始まる) */
    struct ext2_super_block sb; // sbに格納する
    // 1024バイト目から完全に sizeof(sb) バイト読めたときだけ成功
    if (pread(fd, &sb, sizeof(sb), 1024) != (ssize_t)sizeof(sb)) { 
        perror("pread superblock"); return 1;
    }
    if (sb.s_magic != 0xEF53) {
        fprintf(stderr, "ext2 ではありません (magic=0x%04x)\n", sb.s_magic);
        return 1;
    }

    uint32_t block_size = 1024U << sb.s_log_block_size; // データブロックの大きさを計算
    uint32_t inode_size = sb.s_inode_size ? sb.s_inode_size : 128; // inodeの大きさ

    printf("block_size = %u\n", block_size);
    printf("inode_size = %u\n", inode_size);

    /* 2) ブロックグループデスクリプタ読み込み
     *    - block_size==1024 のとき: スーパーブロックがブロック1にあり、
     *      GDT はブロック2 (オフセット2048) から始まる
     *    - block_size>=2048 のとき: スーパーブロックはブロック0内(1024番地)、
     *      GDT はブロック1 (オフセット=block_size) から始まる
     */
    uint64_t gdt_off = (block_size == 1024) ? 2048 : (uint64_t)block_size;
    struct ext2_group_desc gd;
    if (pread(fd, &gd, sizeof(gd), gdt_off) != (ssize_t)sizeof(gd)) {
        perror("pread group desc"); return 1;
    }
    printf("inode_table_block = %u\n", gd.bg_inode_table);

    /* 3) ルートディレクトリは常に inode 番号 2 */
    uint64_t inode_off = (uint64_t)gd.bg_inode_table * block_size
                       + (2 - 1) * inode_size;
    uint8_t *inode_buf = malloc(inode_size);
    if (pread(fd, inode_buf, inode_size, inode_off) != (ssize_t)inode_size) {
        perror("pread inode"); return 1;
    }
    struct ext2_inode *root = (struct ext2_inode *)inode_buf;

    // i_mode:8進数で先頭が04ならディレクトリ
    printf("root i_mode = 0%o, i_size = %u\n", root->i_mode, root->i_size);

    /* 4) ディレクトリのデータブロック(最初の直接ブロック)を読む */
    // ルートディレクトリ（inode 2）が、実際のディレクトリエントリをどのデータブロックに持っているか？
    uint32_t dir_block = root->i_block[0];
    if (dir_block == 0) {
        fprintf(stderr, "ルートディレクトリにデータブロックがありません\n");
        return 1;
    }
    printf("dir_block = %u\n", dir_block);

    uint8_t *block = malloc(block_size);
    if (pread(fd, block, block_size, (uint64_t)dir_block * block_size)
            != (ssize_t)block_size) {
        perror("pread dir block"); return 1;
    }

    /* 5) ディレクトリエントリを順にたどってファイル名を表示 */
    printf("--- root entries ---\n");
    uint32_t off = 0;
    while (off + 8 <= block_size) {
        struct ext2_dir_entry *de = (struct ext2_dir_entry *)(block + off);
        if (de->rec_len == 0) break;

        if (de->inode != 0 && de->name_len > 0) { // inode 番号が 0 → 削除済みエントリ
            uint8_t nlen = de->name_len;
            if (nlen > 255) nlen = 255;
            char name[256];
            memcpy(name, de->name, nlen);
            name[nlen] = '\0';
            printf("  inode=%-6u type=%u name=%s\n",
                   de->inode, de->file_type, name);
        }
        off += de->rec_len;
        if (off >= root->i_size) break;
    }

    free(block);
    free(inode_buf);
    close(fd);
    return 0;
}
```

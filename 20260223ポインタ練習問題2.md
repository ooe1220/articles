```mondai4.c
#include<stdio.h>

int countvalue(int *p, int size,int value){
 int cnt=0;
 while(size != 0){
  if(*p++ == value){
   cnt++;
  }
  size--;
 }
 return cnt;
}

void main(){
 int arr[8]={10,20,30,40,50,30,30,10};
 int *p=&arr;
 int cnt = countvalue(p,8,30);
 printf("cnt=%d\n",cnt);
}
```

```mondai5.c
#include<stdio.h>

char* findchar(char *s, char c){
 char *ret=NULL;
 while(*s!='\0'){
  if(*s==c){
   ret=s;
   break;
  }
  s++;
 }
 return ret;
}

void main(){
 char str[]="aiueo";
 char c = 'e';
 char *ptr = findchar(&str,c);
 printf("*ptr=%c",*ptr);
}
```

# 背景

筆者が`IBM BIOS`のコードを読んでいると気になる処理を発見しました。

https://github.com/gawlas/IBM-PC-BIOS/blob/master/IBM%20PC/PCBIOSV3.ASM

```PCBIOSV3.ASM

;...省略...

TIMER		EQU	40H

;...省略...

;----- INITIALIZE TIMER 1 TO REFRESH MEMORY (タイマー1を初期化し、メモリ・リフレッシュを行う)

C15:					; WRAP_DMA_REG
	MOV	AL,18			; SETUP DIVISOR FOR REFRESH
	OUT	TIMER+1,AL		; WRITE TIMER 1 CNT REG
	OUT	DMA+0DH,AL		; SEND MASTER CLEAR TO DMA

;...省略...

```

`DRAM`は揮発性メモリで、一定の間隔でリフレッシュしないと保存されている値が消えてしまいます。
どうもこの部分でリフレッシュの間隔を定めているようです。
`DRAM`に何か値を書き込んでから、リフレッシュの間隔を広げたらビットが別のものに変わってしまうのではないかと考えて、今回の実験を行いました。


# 手順

1.自作ブートローダを起動する。
2.`DRAM`を崩壊させる処理を書くコードが`DRAM`の上で実行されていては墓穴を掘っているようなものなので`VRAM`上に退避して実行。
以下の記事で機械語を`VRAM`上で動作させることは可能だと証明している。
[8086 VRAMから機械語を実行](https://qiita.com/earthen94/items/12d77dfcb0c9068986e1)
3.`80000`番地に`0x0c41`(VGA mode3でVRAMに書き込むと赤いAになる)を書き込む。
4.リフレッシュの間隔を大きくする
5.少し待つ
6.`80000`番地から値を取り出して`VRAM`に書き込み、赤A以外が表示されるかを確認する

# 結果

失敗に終わりました。DRAMのリフレッシュはDMAに依存していないのでしょうか。。。
失敗作を投稿するか迷いましたが、いつか誰かがもっと良いやり方を教えてくれると信じて上げておきます。

# コード

```boot.asm
[org 0x7C00]
bits 16

start:

    cli
    xor ax, ax
    mov ds, ax
    mov ss, ax
    mov ax, 0xB800
    mov es, ax
    mov si, 0x7C00
    mov di, 0x0000
    mov sp, 0x7C00
    sti
    
    ; ブートローダをVRAMに複製
    mov cx, 512
    cld
    rep movsb ; [ES:DI](B800:0000) ← [DS:SI](0000:7C00)  DI++ SI++

    jmp 0xB800:(continue - start)

continue:

    ; dramのリフレッシュ率を下げる前に値を格納
    ; ES:DI=8000:0000 
    mov ax, 0x8000
    mov es, ax
    xor di, di
    mov ax, 0x0c41 ; 赤A
    mov [es:di], ax
    
    cli ; 割り込み禁止
    ; PIC禁止
    mov al,0xff
    out 0x21, al
    
    mov al, 01110100b   ; Counter 1, Mode 2, LSB/MSB
    out 0x43, al
   
    mov ax, 0xffff
    out 0x41, al
    mov al, ah
    out 0x41, al
    
    mov cx, 0xffff
delay:
    dec cx
    jnz delay
    mov cx, 0xffff
delay2:
    dec cx
    jnz delay2
    mov cx, 0xffff
delay3:
    dec cx
    jnz delay3
    mov cx, 0xffff
delay4:
    dec cx
    jnz delay4
    
    ; dramのリフレッシュ率を下げ後に値が変わっているかを確認
    mov ax, [es:di]
    mov bx, ax ; AX退避
    
    mov ax, 0xB800
    mov es, ax
    
    mov di, 0x0F9C ; VRAM最後から1文字前
    mov [es:di], bx
    add di,2

hlt_loop:
    hlt
    jmp hlt_loop

times 510-($-$$) db 0
dw 0xAA55
```

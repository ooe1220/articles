
# 目的
ARM32のアセンブリを研究していこうと思っています。
Linux上で検証しても良いのですが、筆者はベアメタル環境が好きなので任天堂GBAで実験することにしました。
画面に表示しないとデバッグが困難な為、C言語からアセンブリで書いた関数を呼ぶ形で実行します。
今後困らない為にビルドの手順を記録しておきます。

# C言語からアセンブリを呼ぶ方法

アセンブリに外出しした関数名をC内で明記する
```c
extern int add(int a, int b);
```

今回C言語から呼び出すアセンブリ
```
.syntax unified
    .arch armv7-a
    .arm
    .text
    .global add
    .type add, %function

add:
    add r0, r0, r1     @ r0 = a + b
    bx  lr             @ 返り値

    .size add, .-add
```

# コード

再現可能な様にコード及びコマンドを全掲します。

```bash
rmdir tmp
mkdir tmp

arm-none-eabi-as header.s -o tmp/header.o
arm-none-eabi-as add.s -o tmp/add.o
arm-none-eabi-gcc -c main.c -o tmp/main.o -nostdlib -ffreestanding
arm-none-eabi-gcc -c font.c -o tmp/font.o -nostdlib -ffreestanding
arm-none-eabi-ld tmp/header.o tmp/add.o tmp/main.o tmp/font.o -T link.ld -o tmp/out.elf
arm-none-eabi-objcopy -O binary tmp/out.elf tmp/out.gba
```

.org 0x000000A0
.ascii "GBADEMO     "       @ 名前 12文字
.byte 0x00,0x00,0x00,0x00     @ ゲームコード 4バイト
.byte 0x00,0x00               @ メーカーコード 2バイト
.byte 0x96                    @ 固定値
.byte 0x00                    @ メインユニットコード

.align 2

ResetHandler:
    ldr sp, =0x03007F00

```main.c
extern int add(int a, int b);

#include "font.h"

typedef unsigned char  uint8_t;
typedef unsigned short uint16_t;
typedef unsigned int   uint32_t;

#define REG_DISPCNT (*(volatile uint16_t*)0x04000000) // 画面制御レジスタ
#define VRAM ((volatile uint16_t*)0x06000000)

#define FONT_WIDTH  8
#define FONT_HEIGHT 8
#define CHAR_SPACING 1

#define SCREEN_WIDTH  240
#define SCREEN_HEIGHT 160

//文字座標位置(広域変数がROM領域に置かれてしまい更新不可な為強引な暫定措置)
//作法的に汚いので修正予定。rodataの調整が必要。
volatile int *text_pos_x=(volatile int*)0x03000000;
volatile int *text_pos_y=(volatile int*)0x03000004;

static inline void putpixel(int x, int y, uint16_t c) {
    if (x < 0 || x >= SCREEN_WIDTH ||
        y < 0 || y >= SCREEN_HEIGHT)
        return;
    VRAM[y * SCREEN_WIDTH + x] = c;
}

void printfont(int x, int y, int c, uint16_t color) {
    for (int row = 0; row < FONT_HEIGHT; row++) {
        uint8_t bits = font[c][row];

        for (int col = 0; col < FONT_WIDTH; col++) {
            if (bits & (1 << (7 - col))) {
                putpixel(x + col, y + row, color);
            }
        }
    }
}

/* カーソル付き描画 */
void putchar(char c, uint16_t color) {

    // 改行処理
    if (c == '\n') {
        *text_pos_x = 1;                       // 左端へ
        *text_pos_y += FONT_HEIGHT + 2;        // 1 行下げる
        return;
    }

    printfont(*text_pos_x, *text_pos_y, c, color);
    *text_pos_x += FONT_WIDTH + CHAR_SPACING;
    
    /* 画面右端で折り返す */
    if (*text_pos_x + FONT_WIDTH >= SCREEN_WIDTH) {
        *text_pos_x = 1;
        *text_pos_y += FONT_HEIGHT + 2;
    }
}

void print(const char *str, uint16_t color) {
    while (*str) {
        putchar(*str++, color);
    }
}

void println(const char *str, uint16_t color) {
    print(str, color);
    putchar('\n', color);
}

void print_dec(int value, uint16_t color)
{
    char buf[12]; // 符号 + 最大10桁 + '\0'
    int i = 0;
    int sign = 0;

    // 負数対応
    if (value < 0) {
        sign = 1;
        value = -value;
    }

    // 数値を逆順でバッファに格納
    do {
        buf[i++] = '0' + (value % 10);
        value /= 10;
    } while (value > 0);

    // 符号
    if (sign) {
        buf[i++] = '-';
    }

    // 逆順出力
    while (i > 0) {
        putchar(buf[--i], color);
    }
}

void main(void)
{
    // Mode3 + BG2
    REG_DISPCNT = 0x0403;
    
    volatile int result = add(3, 5);
    
    *text_pos_x=1;
    *text_pos_y=1;
    
    print("RESULT=", 0xFFFF);
    print_dec(result, 0xFFFF);

    while (1);
};
```

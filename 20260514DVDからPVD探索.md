
# やりたいこと
`El Torito no-emulation`形式で、ブートセクタから別のファイルを読み込む為に、第一段階としてLBA16にある`Primary Volume Descriptor / PVD`を読み込む。

# ISOの作成

```
nasm -f bin boot.asm -o boot.bin

mkdir iso
cp boot.bin iso/

xorriso -as mkisofs \
  -o os.iso \
  -b boot.bin \
  -c boot.cat \
  -no-emul-boot \
  -boot-load-size 4 \
  iso
```

# コード

`PVD`先頭の`CD001`と一致したら画面上に赤字で`O`と出す。

```boot.asm
[BITS 16]
[ORG 0x7C00]

start:
    cli
    xor ax, ax
    mov ds, ax
    mov es, ax
    mov ss, ax
    mov sp, 0x7C00
    sti

    ; -----------------------------------
    ; LBA16を読む
    ; INT 13h Extensions (AH=42h)
    ; -----------------------------------

    mov si, dap
    mov ah, 0x42
    int 0x13 ; BIOSがdlにドライブ番号を入れる。絶対に上書きしない
    jc disk_error

    ; -----------------------------------
    ; "CD001"確認
    ; bufferの先頭5バイトを見る
    ; -----------------------------------

    mov si, buffer

    mov al, [si+1]
    cmp al, 'C'
    jne bad_iso

    mov al, [si+2]
    cmp al, 'D'
    jne bad_iso

    mov al, [si+3]
    cmp al, '0'
    jne bad_iso

    mov al, [si+4]
    cmp al, '0'
    jne bad_iso

    mov al, [si+5]
    cmp al, '1'
    jne bad_iso

    ; OK
    jmp success

; -----------------------------------
; failure paths
; -----------------------------------

disk_error:
    mov ax, 0xB800
    mov es, ax
    mov ah, 0x0C ; 赤
    mov al, 'E'
    xor di, di
    mov [es:di], ax
    add di,2
    hlt
    jmp $

bad_iso:
    mov ax, 0xB800
    mov es, ax
    mov ah, 0x0C ; 赤
    mov al, 'B'
    xor di, di
    mov [es:di], ax
    add di,2
    hlt
    jmp $

success:
    mov ax, 0xB800
    mov es, ax
    mov ah, 0x0C ; 赤
    mov al, 'O'
    xor di, di
    mov [es:di], ax
    hlt
    jmp $

; -----------------------------------
; Disk Address Packet (INT 13h 拡張)
; -----------------------------------
dap:
    db 0x10          ; size
    db 0x00          ; reserved
    dw 4             ; number of sectors (ignored for CD in practice)
    dw buffer        ; offset
    dw 0x0000        ; segment
    dq 16            ; LBA = 16

; -----------------------------------
; buffer
; -----------------------------------
buffer:
    times 2048 db 0
```

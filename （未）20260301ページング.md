```
頁数 = page_directory数 × page_table数
          = 1024 × 1024
          = 1,048,576 頁

仮想メモリ空間 = 総頁数 × 1頁の大きさ(4KB)
              = 1,048,576 × 4096 バイト
              = 4,294,967,296 バイト
              = 4GB
```

```kernel.c
typedef unsigned int uint32_t;
uint32_t page_directory[1024] __attribute__((aligned(4096)));
uint32_t page_table0[1024]    __attribute__((aligned(4096)));//0x00000000 ～ 0x003FFFFF(0-4MB)
uint32_t page_table1[1024]    __attribute__((aligned(4096)));//0x00400000 ～ 0x007FFFFF(4-8MB)
#define PAGE_PRESENT (1 << 0)  // bit0: 1=有効, 0=アクセスでページフォルト
#define PAGE_RW      (1 << 1)  // bit1: 1=書き込み可能, 0=読み取り専用

void paging_init()
{
    // --- 全PDEを無効 ---
    for (int i = 0; i < 1024; i++) {
        page_directory[i] = 0;
    }

    for (int i = 0; i < 1024; i++) {
        page_table0[i] = (i * 0x1000) | PAGE_PRESENT | PAGE_RW; // 0 物理=仮想
        page_table1[i] = 0; // 4～8MB (丁度4MB目しか使わないので全部0にする)
    }
    page_directory[0] = ((uint32_t)page_table0) | PAGE_PRESENT | PAGE_RW;
    page_directory[1] = ((uint32_t)page_table1) | PAGE_PRESENT | PAGE_RW;
   
    // 仮想 0x400000 → 物理 0xB8000 (メモリに書き込んだことが分かり易い様にVRAMへ投射する)
    page_table1[0] = 0xB8000 | PAGE_PRESENT | PAGE_RW;
    
        // --- CR3 ---
    asm volatile("mov %0, %%cr3" :: "r"(page_directory));

    // --- ページング有効 ---
    uint32_t cr0;
    asm volatile("mov %%cr0, %0" : "=r"(cr0));
    cr0 |= 0x80000000; //PGビットを立てる 0x80000000 = 10000000 00000000 00000000 00000000
    asm volatile("mov %0, %%cr0" :: "r"(cr0));
}

void kernel_main() {
    
    paging_init();
    
    // 仮想アドレス
    volatile unsigned short* v = (unsigned short*)0x400000;

    v[0] = 0x0F41; // 'A'
    v[1] = 0x0F42; // 'B'
    v[2] = 0x0F43; // 'C'

    while(1);
}
```

```Program.cs
using System;
using System.Drawing;
using System.Windows.Forms;

namespace SearchApp
{
    public class SearchForm : Form
    {
        // 部品定義
        TextBox searchTextBox;
        Button searchButton;
        
        public SearchForm()
        {
            InitializeForm();
            InitializeControls();
        }
        
        private void InitializeForm()
        {
            // Windowsの設定
            this.Text = "検索機能";
            this.Size = new Size(400, 150);
            this.StartPosition = FormStartPosition.CenterScreen;
            this.FormBorderStyle = FormBorderStyle.FixedSingle;
            this.MaximizeBox = false; // trueで最大化可能
        }
        
        private void InitializeControls()
        {
            // テキストボックス生成
            searchTextBox = new TextBox
            {
                Location = new Point(20, 30),
                Size = new Size(260, 25),
                Font = new Font("Microsoft YaHei", 10),
                PlaceholderText = "入力..."
            };
            
            // ボタン生成
            searchButton = new Button
            {
                Location = new Point(290, 28),
                Size = new Size(80, 30),
                Text = "検索",
                Font = new Font("Microsoft YaHei", 10),
                BackColor = Color.FromArgb(0, 120, 215),
                ForeColor = Color.White,
                FlatStyle = FlatStyle.Flat
            };
            searchButton.FlatAppearance.BorderSize = 0;
            
            // ボタンにイベント追加
            searchButton.Click += SearchButton_Click;
            
            // テキストエンター押下時のイベント追加
            searchTextBox.KeyDown += (sender, e) =>
            {
                if (e.KeyCode == Keys.Enter)
                {
                    PerformSearch();
                }
            };
            
            // 部品をWindowへ追加
            this.Controls.Add(searchTextBox);
            this.Controls.Add(searchButton);
        }
        
        private void SearchButton_Click(object sender, EventArgs e)
        {
            PerformSearch();
        }
        
        private void PerformSearch()
        {
            string keyword = searchTextBox.Text.Trim();
            
            if (string.IsNullOrEmpty(keyword))
            {
                MessageBox.Show("空です", "結果", 
                    MessageBoxButtons.OK, MessageBoxIcon.Information);
                return;
            }
            
            MessageBox.Show($"入力: {keyword}", "結果", 
                MessageBoxButtons.OK, MessageBoxIcon.Information);
        }
    }
    
    public class Program
    {
        [STAThread]
        static void Main()
        {
            //Application.EnableVisualStyles();
            //Application.SetCompatibleTextRenderingDefault(false);
            Application.Run(new SearchForm());
        }
    }
}
```

# 仕組み

Windows

`Form`を継承する。
```cs
public class SearchForm : Form
```

`this.XXXX`に対してWindowの設定をする。

## 各部品

`TextBox`,`Button`等、部品の種類毎に複数のクラスが用意されている。

以下のように設定値を引数としてインスタンスを生成
```cs
            searchTextBox = new TextBox
            {
                Location = new Point(20, 30),
                Size = new Size(260, 25),
                Font = new Font("Microsoft YaHei", 10),
                PlaceholderText = "入力..."
            };
```

## イベントを追加

### 別の場所で定義したメソッドを登録する例

```cs
searchButton.Click += SearchButton_Click;
```

```cs
        private void SearchButton_Click(object sender, EventArgs e)
        {
            PerformSearch();
        }
```

### 代入する時に直接処理を書く例

```cs
            searchTextBox.KeyDown += (sender, e) =>
            {
                if (e.KeyCode == Keys.Enter)
                {
                    PerformSearch();
                }
            };
```

## Window上に各部品を追加

```cs
            this.Controls.Add(searchTextBox);
            this.Controls.Add(searchButton);
```

## Windowを画面上に表示する

`Form`を継承して作ったクラスのインスタンスを生成して渡す。
```cs
Application.Run(new SearchForm());
```

## 全体の流れ
1.Formを継承したクラス作成
2.各部品のインスタンス化と設定
3.イベントハンドラの設定
4.各部品ををForm(Windowのインスタンス)に追加
5.Application.Run()で実行

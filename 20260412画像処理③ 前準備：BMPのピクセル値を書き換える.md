# C言語処理
BMP形式の画像を読み込み、ピクセルの値を書き換えてBMP形式で書き出す雛形を書きました。

```main.c
// gcc main.c -o main.exe
#include <stdio.h>
#include <stdlib.h>

#pragma pack(1)

typedef struct {
    unsigned short bfType;
    unsigned int   bfSize;
    unsigned short bfReserved1;
    unsigned short bfReserved2;
    unsigned int   bfOffBits;
} BITMAPFILEHEADER;

typedef struct {
    unsigned int   biSize;
    int            biWidth;
    int            biHeight;
    unsigned short biPlanes;
    unsigned short biBitCount;
    unsigned int   biCompression;
    unsigned int   biSizeImage;
    int            biXPelsPerMeter;
    int            biYPelsPerMeter;
    unsigned int   biClrUsed;
    unsigned int   biClrImportant;
} BITMAPINFOHEADER;

#pragma pack()

int main() {
    FILE *fp = fopen("input.bmp", "rb");
    if (!fp) {
        printf("input.bmp 開けない\n");
        return 1;
    }

    BITMAPFILEHEADER fileHeader;
    BITMAPINFOHEADER infoHeader;

    fread(&fileHeader, sizeof(fileHeader), 1, fp);
    fread(&infoHeader, sizeof(infoHeader), 1, fp);

    // チェック
    if (fileHeader.bfType != 0x4D42) {
        printf("BMPでない\n");
        return 1;
    }

    if (infoHeader.biBitCount != 24) {
        printf("24bit限定\n");
        return 1;
    }

    int width  = infoHeader.biWidth;
    int height = infoHeader.biHeight;

    // 4バイト境界
    int rowSize = ((width * 3 + 3) / 4) * 4;

    // メモリ確保(ピクセルのバイナリ)
    unsigned char *data = malloc(rowSize * height);
    if (!data) {
        printf("メモリ確保失敗\n");
        return 1;
    }

    // ピクセルへ移動
    fseek(fp, fileHeader.bfOffBits, SEEK_SET);

    fread(data, 1, rowSize * height, fp);
    fclose(fp);

    // =========================
    // 画素操作（ここが本体）
    // =========================

    for (int y = 0; y < height; y++) {
        for (int x = 0; x < width; x++) {

            // ★ここが重要（上下反転して読む）
            unsigned char *p = data + (height - 1 - y) * rowSize + x * 3;

            // BGR
            int b = p[0];
            int g = p[1];
            int r = p[2];

            // 明るくする
            //b += 50;
            //g += 50;
            //r += 50;

            // 最大値超えたら白に固定
            //if (b > 255) b = 255;
            //if (g > 255) g = 255;
            //if (r > 255) r = 255;

            // 例2：灰色諧調
            int gray = (r + g + b) / 3;
            r = g = b = gray;
        	
            p[0] = b;
            p[1] = g;
            p[2] = r;
        }
    }


    // =========================
    // 書き出し
    // =========================

    FILE *out = fopen("output.bmp", "wb");
    if (!out) {
        printf("output.bmp 開けない\n");
        free(data);
        return 1;
    }

    fwrite(&fileHeader, sizeof(fileHeader), 1, out);
    fwrite(&infoHeader, sizeof(infoHeader), 1, out);
    fwrite(data, 1, rowSize * height, out);

    fclose(out);
    free(data);

    printf("output.bmp 生成完了\n");

    return 0;
}
```

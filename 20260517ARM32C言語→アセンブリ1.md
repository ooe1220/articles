C言語で書いても問題無く動いていますが、実際CPU上でどう動いているのか気になってきたので確認することにしました。
GCCが生成したアセンブリは長いので必要な部分だけ載せます。


動作確認様のスクリプト:
```
arm-none-eabi-as header.s -o header.o
arm-none-eabi-gcc -c main.c -o main.o -nostdlib -ffreestanding
arm-none-eabi-ld header.o main.o -T link.ld -o out.elf
arm-none-eabi-objcopy -O binary out.elf out.gba
```

アセンブリを見るスクリプト:
```
arm-none-eabi-gcc -S main.c -o main.s -nostdlib -ffreestanding
```

# VRAM書き込み部分

```main.c
typedef unsigned char  uint8_t;
typedef unsigned short uint16_t;
typedef unsigned int   uint32_t;

#define REG_DISPCNT (*(volatile uint16_t*)0x04000000) // 画面制御レジスタ
#define VRAM ((volatile uint16_t*)0x06000000)

#define SCREEN_WIDTH  240
#define SCREEN_HEIGHT 160

void main(void)
{
    // Mode3 + BG2
    REG_DISPCNT = 0x0403;
	
    VRAM[240] = 0xFFFF;
    VRAM[241] = 0xFFFF;
    VRAM[242] = 0xFFFF;
    VRAM[243] = 0xFFFF;

    while (1);
};
```

```main.s
	str	fp, [sp, #-4]!
	add	fp, sp, #0
	mov	r3, #67108864   @ 67108864=0x04000000=画面制御レジスタアドレス
	ldr	r2, .L3         @ 1027=0x0403=Mode3 + BG2の設定値
	strh	r2, [r3]	@ アドレス0x04000000=1027
	ldr	r3, .L3+4       @ VRAMアドレス
	mvn	r2, #0          @ 即値 0 のビット反転を代入 r2 = 0xFFFFFFFF
	strh	r2, [r3]	@ アドレス100663776=0xFFFF(strhなので16bitのみ)
	ldr	r3, .L3+8
	mvn	r2, #0
	strh	r2, [r3]	@ movhi
	ldr	r3, .L3+12
	mvn	r2, #0
	strh	r2, [r3]	@ movhi
	ldr	r3, .L3+16
	mvn	r2, #0
	strh	r2, [r3]	@ movhi
.L2:
	b	.L2
.L4:
	.align	2
.L3:
	.word	1027
	.word	100663776
	.word	100663778
	.word	100663780
	.word	100663782
```


# 文字列を関数に渡す

```main.c
typedef unsigned char  uint8_t;
typedef unsigned short uint16_t;
typedef unsigned int   uint32_t;

#define REG_DISPCNT (*(volatile uint16_t*)0x04000000) // 画面制御レジスタ
#define VRAM ((volatile uint16_t*)0x06000000)

#define SCREEN_WIDTH  240
#define SCREEN_HEIGHT 160

void strcnt(const char *str) {
    int cnt=0;
    while (*str++) {
        cnt++;
    }
}

void main(void)
{
    // Mode3 + BG2
    REG_DISPCNT = 0x0403;
    
    strcnt("TEST STR");																																																			
	
    while (1);
};
```

```main.s
strcnt:
	str	fp, [sp, #-4]!
	add	fp, sp, #0
	sub	sp, sp, #20
	str	r0, [fp, #-16]
	mov	r3, #0
	str	r3, [fp, #-8]   @ int cnt=0;
	b	.L2
	
.L3:    @ while内の処理 
	ldr	r3, [fp, #-8]   @ r3にcnt変数格納
	add	r3, r3, #1      @ cnt++
	str	r3, [fp, #-8]   @ cnt変数をスタックに戻す
.L2:
	ldr	r3, [fp, #-16]  @ r3 = str
	add	r2, r3, #1	@ str++ （ポインタ更新）
	str	r2, [fp, #-16]
	ldrb	r3, [r3]	@ r3 = *str（更新前の値）
	cmp	r3, #0
	bne	.L3             @ if(*str != '\0')ループ本体へ
	nop
	nop
	add	sp, fp, #0
	@ sp needed
	ldr	fp, [sp], #4
	bx	lr
	.size	strcnt, .-strcnt
	.section	.rodata
	.align	2
```


```main.s
.LC0:
	.ascii	"TEST STR\000"
	
main:
	push	{fp, lr}
	add	fp, sp, #4
	mov	r3, #67108864
	ldr	r2, .L6
	strh	r2, [r3]	@ movhi
	
        @ strcnt("TEST STR");
	ldr	r0, .L6+4       @ "TEST STR\000"を引数に設定
	bl	strcnt          @ strcnt呼び出し
.L5:
	b	.L5
.L7:
	.align	2
.L6:
	.word	1027
	.word	.LC0
	.size	main, .-main
```


```kernel.c
typedef unsigned int  uint32_t;
typedef unsigned char uint8_t;


/* 画面出力 */
void putc(char c)
{
    static int pos = 0;
    volatile unsigned short* vram = (unsigned short*)0xB8000;

    vram[pos++] = 0x0F00 | c;
    
    if (pos >= 2000) // VRAMはみ出し防止
        pos = 0;
}


/* task 構造体 */

typedef struct {
    uint32_t esp;
} task_t;


task_t tasks[2];
int current = 0;


/* スタック */
uint8_t stackA[4096];
uint8_t stackB[4096];


/* context switch */

__attribute__((naked)) void context_switch(uint32_t *old, uint32_t new)
{
    __asm__ volatile(

        "pusha\n"

        "mov 36(%esp), %eax\n" // eax = &tasks[prev].esp (old)
        "mov %esp, (%eax)\n"   // *tasks[prev].esp = esp

        "mov 40(%esp), %eax\n" // eax = tasks[next].esp (new)
        "mov %eax, %esp\n"     // esp = tasks[next].esp

        "popa\n"
        "ret\n"
    );
}


/* スケジューラ */
void yield()
{
    int next = (current + 1) % 2;

    int prev = current;
    current = next;

    context_switch(&tasks[prev].esp, tasks[next].esp);
}

/* タスク */
void taskA()
{
    while (1) {
        putc('A');
        yield();
    }
}

void taskB()
{
    while (1) {
        putc('B');
        yield();
    }
}

/* タスク初期化 
 CPUが途中から再開できるスタックを人工的に作る 
*/
void task_init()
{
    uint32_t *sp;

    sp = (uint32_t*)(stackA + 4096);

    *--sp = (uint32_t)taskA; // 返りアドレス
    *--sp = 0; // eax  popaの順番で積む
    *--sp = 0; // ecx
    *--sp = 0; // edx
    *--sp = 0; // ebx
    *--sp = 0; // esp(dummy)
    *--sp = 0; // ebp
    *--sp = 0; // esi
    *--sp = 0; // edi

    tasks[0].esp = (uint32_t)sp;

    sp = (uint32_t*)(stackB + 4096);

    *--sp = (uint32_t)taskB;
    *--sp = 0;
    *--sp = 0;
    *--sp = 0;
    *--sp = 0;
    *--sp = 0;
    *--sp = 0;
    *--sp = 0;
    *--sp = 0;

    tasks[1].esp = (uint32_t)sp;
}

/* カーネル入口 */

void kernel_main()
{
    uint32_t dummy;

    task_init();

    current = 0;

    context_switch(&dummy, tasks[0].esp);

    while (1);
}
```

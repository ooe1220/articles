```test4.asm
.model small
.stack 100h
.data
value db 12h,34h,56h,78h,9ah,0bch,0deh,0f1h,12h,34h,56h,78h,9ah,0bch,0deh,0f1h

.code
main proc

 ;VRAM=ES:DI
 mov ax,0b800h
 mov es,ax
 mov di,160 ; (80*1+0)+2 zahyo(2,1)

 ;hyouji suru atai
 mov ax,@data
 mov ds,ax
 lea si,value

 mov cx,10
dump_loop: 
 call SIDump
 call Memdump
 call NewLine
 loop dump_loop

 mov ah,4ch
 int 21h
main endp

;----------------------------
; NewLine
;----------------------------
NewLine proc
 xor dx,dx
 mov ax,di
 mov bx,160
 div bx     ; AX/BX=AX...DX
 mov bx,160
 mul bx     ; DX:AX=AX*BX
 add ax,160
 mov di,ax
 ret
NewLine endp


;----------------------------
; SIDump 
; input  : SI
; output : VRAM
;----------------------------
SIDump proc
 push cx
 mov bx,si

 ; 1ketame
 mov ax,bx
 mov cl,12
 shr ax,cl
 call Hex1
 mov byte ptr es:[di],al
 mov byte ptr es:[di+1],07h
 add di,2

 ; 2ketame
 mov ax,bx
 mov cl,8
 shr ax,cl
 and ax,0fh
 call Hex1
 mov byte ptr es:[di],al
 mov byte ptr es:[di+1],07h
 add di,2

 ; 3ketame
 mov ax,bx
 mov cl,4
 shr ax,cl
 and ax,0fh
 call Hex1
 mov byte ptr es:[di],al
 mov byte ptr es:[di+1],07h
 add di,2

 ; 4ketame
 mov ax,bx
 mov cl,0
 shr ax,cl
 and ax,0fh
 call Hex1
 mov byte ptr es:[di],al
 mov byte ptr es:[di+1],07h
 add di,2

 ; space
 mov al,' '
 mov ah,07h
 mov byte ptr es:[di],al
 mov byte ptr es:[di+1],07h
 add di,2
 pop cx
 ret
SIDump endp

;----------------------------
;Memdump
; input  : SI
; output : DI,VRAM
;----------------------------
Memdump proc
  push cx
  mov cx,16
next_byte:
 mov bl,ds:[si]

 ;kai 4bit
 push cx
 mov al,bl
 mov cl,4
 shr al,cl 
 and al,0fh
 call Hex1
 mov byte ptr es:[di],al
 mov byte ptr es:[di+1],07h
 add di,2
 pop cx

 ;joui 4bit
 mov al,bl
 and al,0fh
 call Hex1
 mov byte ptr es:[di],al
 mov byte ptr es:[di+1],07h
 add di,2

 ; space
 mov al,' '
 mov ah,07h
 mov byte ptr es:[di],al
 mov byte ptr es:[di+1],07h
 add di,2

 inc si
 loop next_byte
 pop cx
 ret
Memdump endp

;----------------------------
; input  : AL
; output : AL(ASCII 0-F)
;----------------------------
Hex1 proc
 cmp al,9
 jbe digit
 add al,7 

 digit:
  add al,'0'
  ret
Hex1 endp

end main
```

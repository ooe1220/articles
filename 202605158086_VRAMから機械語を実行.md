# 目的
`VRAM(0xB8000)`に機械語を書き込み、そこへ`jmp`して正常に実行されるのかを検証します。
VRAMは本来画面に表示する為に使用する為のもので、無理やり`CS:IP`でVRAM領域を指して実行することは、インテルも考えていなかったのでは無いでしょうか。(知りませんが)

# 検証

```bash
nasm -f bin boot.asm -o boot.bin
```

# コード

1. 先ずはBIOSによってこのブートローダが`0x7C00`に読み込まれます
2. ブートローダ自身をVRAM(0xB8000〜)に複製する
3. `jmp`命令でVRAMへ跳ぶ
4. 画面右下に`OK`と表示されるかを観察します(先頭には機械語が入っているので避ける)

```boot.asm
[org 0x7C00]
bits 16

start:

    cli
    xor ax, ax
    mov ds, ax
    mov ss, ax
    mov ax, 0xB800
    mov es, ax
    mov si, 0x7C00
    mov di, 0x0000
    mov sp, 0x7C00
    sti
    
    ; ブートローダをVRAMに複製
    mov cx, 512
    cld
    rep movsb ; [ES:DI](B800:0000) ← [DS:SI](0000:7C00)  DI++ SI++

    jmp 0xB800:(continue - start)

continue:

    mov ax, 0xB800
    mov es, ax
    mov di, 0x0F9C ; VRAM最後から2文字分
    mov ah, 0x0C ; 赤
    mov al, 'O'
    mov [es:di], ax
    add di,2
    mov al, 'K'
    mov [es:di], ax
    add di,2

hlt_loop:
    hlt
    jmp hlt_loop

times 510-($-$$) db 0
dw 0xAA55
```

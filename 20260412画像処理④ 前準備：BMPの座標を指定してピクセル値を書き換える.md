```main.c
// gcc main.c -o main.exe
#include <stdio.h>
#include <stdlib.h>

#pragma pack(1)

typedef struct {
    unsigned short bfType;
    unsigned int   bfSize;
    unsigned short bfReserved1;
    unsigned short bfReserved2;
    unsigned int   bfOffBits;
} BITMAPFILEHEADER;

typedef struct {
    unsigned int   biSize;
    int            biWidth;
    int            biHeight;
    unsigned short biPlanes;
    unsigned short biBitCount;
    unsigned int   biCompression;
    unsigned int   biSizeImage;
    int            biXPelsPerMeter;
    int            biYPelsPerMeter;
    unsigned int   biClrUsed;
    unsigned int   biClrImportant;
} BITMAPINFOHEADER;

#pragma pack()

// =========================
// Image構造体
// =========================
typedef struct {
    unsigned char *data;
    int width;
    int height;
    int rowSize;
} Image;

// =========================
// 座標 → ピクセル取得
// =========================
unsigned char* get_pixel(Image *img, int x, int y) {
    if (x < 0 || x >= img->width || y < 0 || y >= img->height) return NULL;

    return img->data + (img->height - 1 - y) * img->rowSize + x * 3;
}

// =========================
// 色操作
// =========================
void set_rgb(unsigned char *p, int r, int g, int b) {
    p[0] = b;
    p[1] = g;
    p[2] = r;
}

void to_gray(unsigned char *p) {
    int b = p[0];
    int g = p[1];
    int r = p[2];

    int gray = (r + g + b) / 3;

    p[0] = gray;
    p[1] = gray;
    p[2] = gray;
}

// =========================
// 点描画
// =========================
void draw_point(Image *img, int x, int y, int r, int g, int b) {
    unsigned char *p = get_pixel(img, x, y);
    if (p) set_rgb(p, r, g, b);
}

// =========================
// 矩形塗り潰し
// =========================
void fill_rect(Image *img, int x0, int y0, int w, int h, int r, int g, int b) {
    for (int y = y0; y < y0 + h; y++) {
        for (int x = x0; x < x0 + w; x++) {
            draw_point(img, x, y, r, g, b);
        }
    }
}

// =========================
// 全体グレースケール
// =========================
void apply_gray(Image *img) {
    for (int y = 0; y < img->height; y++) {
        for (int x = 0; x < img->width; x++) {
            unsigned char *p = get_pixel(img, x, y);
            if (p) to_gray(p);
        }
    }
}

// =========================
// main
// =========================
int main() {
    FILE *fp = fopen("input.bmp", "rb");
    if (!fp) {
        printf("input.bmp 開けない\n");
        return 1;
    }

    BITMAPFILEHEADER fileHeader;
    BITMAPINFOHEADER infoHeader;

    fread(&fileHeader, sizeof(fileHeader), 1, fp);
    fread(&infoHeader, sizeof(infoHeader), 1, fp);

    if (fileHeader.bfType != 0x4D42) {
        printf("BMPでない\n");
        return 1;
    }

    if (infoHeader.biBitCount != 24) {
        printf("24bit限定\n");
        return 1;
    }

    Image img;

    img.width  = infoHeader.biWidth;
    img.height = infoHeader.biHeight;
    img.rowSize = ((img.width * 3 + 3) / 4) * 4;

    img.data = malloc(img.rowSize * img.height);
    if (!img.data) {
        printf("メモリ確保失敗\n");
        return 1;
    }

    fseek(fp, fileHeader.bfOffBits, SEEK_SET);
    fread(img.data, 1, img.rowSize * img.height, fp);
    fclose(fp);

    // =========================
    // 座標を指定して操作
    // =========================

    // ① 全体グレースケール
    apply_gray(&img);

    // ② 点を打つ
    draw_point(&img, 100, 50, 255, 0, 0); // 赤

    // ③ 矩形描画
    fill_rect(&img, 150, 100, 100, 80, 0, 255, 0); // 緑

    // ④ 任意領域だけグレースケール
    for (int y = 200; y < 300; y++) {
        for (int x = 200; x < 300; x++) {
            unsigned char *p = get_pixel(&img, x, y);
            if (p) to_gray(p);
        }
    }

    // =========================
    // 書き出し
    // =========================

    FILE *out = fopen("output.bmp", "wb");
    if (!out) {
        printf("output.bmp 開けない\n");
        free(img.data);
        return 1;
    }

    fwrite(&fileHeader, sizeof(fileHeader), 1, out);
    fwrite(&infoHeader, sizeof(infoHeader), 1, out);
    fwrite(img.data, 1, img.rowSize * img.height, out);

    fclose(out);
    free(img.data);

    printf("output.bmp 生成完了\n");

    return 0;
}
```

```LocalOutlookReader.cs
using System;
using System.Runtime.InteropServices;

public class LocalOutlookReader
{
    public void ReadInboxEmails()
    {
        try
        {
            
            // Outlookオブジェクト
            dynamic outlookApp = Activator.CreateInstance(
                Type.GetTypeFromProgID("Outlook.Application"));
            
            dynamic mapiNamespace = outlookApp.GetNamespace("MAPI");
            
            // 受信箱取得
            dynamic inbox = mapiNamespace.GetDefaultFolder(6);  // 6 = 
            
            // メール一覧取得
            dynamic items = inbox.Items;
            
            // 受信時間で並び替え
            items.Sort("[ReceivedTime]", true);
            
            Console.WriteLine("\n=== Type情報 ===");
            Console.WriteLine($"ソフト名: {outlookApp.Name}");
            Console.WriteLine($"製品番号: {outlookApp.ProductCode}");
            Console.WriteLine($"バージョン: {outlookApp.Version}");
            
            Console.WriteLine("\n=== Namespace情報 ===");
            Console.WriteLine($"Type: {mapiNamespace.GetType().FullName}");
            Console.WriteLine($"CurrentUser.Name: {mapiNamespace.CurrentUser.Name}");
            Console.WriteLine($"CurrentUser.Address: {mapiNamespace.CurrentUser.Address}");
            Console.WriteLine($"CurrentUser.AddressEntry.Type: {mapiNamespace.CurrentUser.AddressEntry.Type}");
            Console.WriteLine($"Stores.Count: {mapiNamespace.Stores.Count}");
            Console.WriteLine($"DefaultStore.DisplayName: {mapiNamespace.DefaultStore?.DisplayName ?? "N/A"}");
            Console.WriteLine($"ExchangeMailboxServerName: {mapiNamespace.ExchangeMailboxServerName ?? "N/A"}");
            Console.WriteLine($"ExchangeMailboxServerVersion: {mapiNamespace.ExchangeMailboxServerVersion ?? "N/A"}");
            Console.WriteLine($"Offline: {mapiNamespace.Offline}");
            
            Console.WriteLine("\n=== inbox情報 ===");
            Console.WriteLine($"Name: {inbox.Name}");
            Console.WriteLine($"DefaultItemType: {inbox.DefaultItemType}");
            Console.WriteLine($"Description: {inbox.Description}");
            Console.WriteLine($"EntryID: {inbox.EntryID}");
            Console.WriteLine($"StoreID: {inbox.StoreID}");
            Console.WriteLine($"FolderPath: {inbox.FolderPath}");
            
            Console.WriteLine($"受信箱計 {items.Count} 件");
            Console.WriteLine("\n=== 最新の5件を表示 ===");
            
            int count = 0;
            foreach (dynamic mail in items)
            {
                if (count >= 5) break;
                
                Console.WriteLine($"差出人: {mail.SenderName}");
                Console.WriteLine($"表題: {mail.Subject}");
                Console.WriteLine($"時間: {mail.ReceivedTime}");
                Console.WriteLine($"状況: {(mail.UnRead ? "未読" : "既読")}");
                Console.WriteLine(new string('-', 40));
                
                count++;
            }
            
            // COM解放
            Marshal.ReleaseComObject(items);
            Marshal.ReleaseComObject(inbox);
            Marshal.ReleaseComObject(mapiNamespace);
            Marshal.ReleaseComObject(outlookApp);
        }
        catch (Exception ex)
        {
            Console.WriteLine($"err: {ex.Message}");
        }
    }
}
```

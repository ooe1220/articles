typedef unsigned int  uint32_t;
typedef unsigned char uint8_t;

uint8_t font8x8[256][8] = {

    ['A'] = {
        0b00111100,
        0b01000010,
        0b01000010,
        0b01111110,
        0b01000010,
        0b01000010,
        0b01000010,
        0b00000000
    },

    ['B'] = {
        0b01111100,
        0b01000010,
        0b01000010,
        0b01111100,
        0b01000010,
        0b01000010,
        0b01111100,
        0b00000000
    }

};

void putpixel(int x, int y, uint8_t color) {
    volatile uint8_t* vram = (uint8_t*)0xA0000;
    vram[y*320 + x] = color;
}

void draw_char(char c, int x, int y, uint8_t color) {
    for (int row = 0; row < 8; row++) {
        uint8_t bits = font8x8[c][row];
        for (int col = 0; col < 8; col++) {
            if (bits & (1 << (7 - col)))
                putpixel(x + col, y + row, color);
        }
    }
}

void draw_string(const char* s, int x, int y, uint8_t color) {
    while (*s) {
        draw_char(*s++, x, y, color);
        x += 8; // 文字幅
    }
}

void kernel_main()
{
    draw_string("AB", 10, 20, 15);
    
    asm volatile("cli"); 
    asm volatile("hlt"); 
}

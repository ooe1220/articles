
# 初めに

筆者が使用している任天堂GBAは2000年くらい、まだ保育所に通っていた頃に買って貰ったものです。
長年殆ど起動もせず押入れに入れっぱなしにしていました。
メモリは大量の半導体で出来ていますが、こんなにも放置していたら記憶素子が数個くらいは壊れているのではないかと不安になりました。
そこで自分でコードを書き、壊れていないか調べてみることにしました。

# 方針

任天堂GBAには高速で容量の少ないメモリ`IWRAM(32KB)`と少し遅いくて大容量のメモリ`EWRAM(256KB)`があります。
この2つのメモリに対して値を書き込んだ後に、書き込んだ値とメモリから読み込んだ値を比較して一致するかを調べます。
一致していなければ値を保持できていないので記憶素子が破損している可能性があります。
どこかのビットが破損している状態で0等を返して偶然一致してしまうことを防ぐ為に以下の値で全ビットに対し`0`と`1`の両方を書き込みます。
具体的には以下の値をそれぞれ書き込んで検証します。

`0x0000 = 0000 0000 0000 0000`
`0xFFFF = 1111 1111 1111 1111`
`0xAAAA = 1010 1010 1010 1010`
`0x5555 = 0101 0101 0101 0101`

`puts`等をまだ用意していない為、結果毎に画面の色を変えて可視化します。
EWRAMに故障箇所があれば緑
IWRAMに故障箇所があれば赤
問題が無ければ青

# 検証用コード

ビルドの手順は筆者の以下の記事を参考
[GBA自作コードを実機で動作させる手順](https://qiita.com/earthen94/items/6ca53e74ea3833c63575)

```main.c
typedef unsigned short uint16_t;
typedef unsigned int   uint32_t;

#define REG_DISPCNT (*(volatile uint16_t*)0x04000000) // 画面制御レジスタ

#define EWRAM_BASE 0x02000000
#define IWRAM_BASE 0x03000000
#define EWRAM_SIZE (256 * 1024)
#define IWRAM_SAFE_SIZE (28 * 1024)  // 上4KBはスタック用に空ける 本来は(32 * 1024)

#define VRAM ((volatile uint16_t*)0x06000000)

void vram_error(uint16_t color)
{
    for (int i = 0; i < 240 * 160; i++) {
        VRAM[i] = color;
    }

    while (1);
}

void ewram_test(uint16_t pattern)
{
    volatile uint16_t *p = (volatile uint16_t *)EWRAM_BASE;

    for (uint32_t i = 0; i < EWRAM_SIZE / 2; i++) {
        p[i] = pattern;
    }

    for (uint32_t i = 0; i < EWRAM_SIZE / 2; i++) {
        if (p[i] != pattern) {
            vram_error(0x07C0); // EWRAMなら緑
        }
    }
}

void iwram_test(uint16_t pattern)
{
    volatile uint16_t *p = (volatile uint16_t *)IWRAM_BASE;

    for (uint32_t i = 0; i < IWRAM_SAFE_SIZE / 2; i++) {
        p[i] = pattern;
    }

    for (uint32_t i = 0; i < IWRAM_SAFE_SIZE / 2; i++) {
        if (p[i] != pattern) {
            vram_error(0x001F); // IWRAMなら赤
        }
    }
}

void main(void)
{
    // Mode3 + BG2
    REG_DISPCNT = 0x0403;
	
    ewram_test(0x0000);
    ewram_test(0xFFFF);
    ewram_test(0xAAAA);
    ewram_test(0x5555);

    iwram_test(0x0000);
    iwram_test(0xFFFF);
    iwram_test(0xAAAA);
    iwram_test(0x5555);
    
    vram_error(0x7C00);

    while (1);
}
```

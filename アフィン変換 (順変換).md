# 実験用コード

`ここに個別の実験用コードを配置`の部分を下に掲載しているコードに置き換える。

<summary>main.c</summary>

```main.c
// gcc main.c -o main.exe
#include <stdio.h>
#include <stdlib.h>
#include <math.h>

#pragma pack(1)

typedef struct {
    unsigned short bfType;
    unsigned int   bfSize;
    unsigned short bfReserved1;
    unsigned short bfReserved2;
    unsigned int   bfOffBits;
} BITMAPFILEHEADER;

typedef struct {
    unsigned int   biSize;
    int            biWidth;
    int            biHeight;
    unsigned short biPlanes;
    unsigned short biBitCount;
    unsigned int   biCompression;
    unsigned int   biSizeImage;
    int            biXPelsPerMeter;
    int            biYPelsPerMeter;
    unsigned int   biClrUsed;
    unsigned int   biClrImportant;
} BITMAPINFOHEADER;

#pragma pack()

// =========================
// Image構造体
// =========================
typedef struct {
    unsigned char *data;
    int width;
    int height;
    int rowSize;
} Image;

// =========================
// 座標 → ピクセル取得
// =========================
unsigned char* get_pixel(Image *img, int x, int y) {
    if (x < 0 || x >= img->width || y < 0 || y >= img->height) return NULL;

    return img->data + (img->height - 1 - y) * img->rowSize + x * 3;
}

// =========================
// 色操作
// =========================
void set_rgb(unsigned char *p, int r, int g, int b) {
    p[0] = b;
    p[1] = g;
    p[2] = r;
}

void to_gray(unsigned char *p) {
    int b = p[0];
    int g = p[1];
    int r = p[2];

    int gray = (r + g + b) / 3;

    p[0] = gray;
    p[1] = gray;
    p[2] = gray;
}

// =========================
// 点描画
// =========================
void draw_point(Image *img, int x, int y, int r, int g, int b) {
    unsigned char *p = get_pixel(img, x, y);
    if (p) set_rgb(p, r, g, b);
}

// =========================
// 矩形塗り潰し
// =========================
void fill_rect(Image *img, int x0, int y0, int w, int h, int r, int g, int b) {
    for (int y = y0; y < y0 + h; y++) {
        for (int x = x0; x < x0 + w; x++) {
            draw_point(img, x, y, r, g, b);
        }
    }
}

// =========================
// 全体グレースケール
// =========================
void apply_gray(Image *img) {
    for (int y = 0; y < img->height; y++) {
        for (int x = 0; x < img->width; x++) {
            unsigned char *p = get_pixel(img, x, y);
            if (p) to_gray(p);
        }
    }
}

// =========================
// main
// =========================
int main() {
    FILE *fp = fopen("input.bmp", "rb");
    if (!fp) {
        printf("input.bmp 開けない\n");
        return 1;
    }

    BITMAPFILEHEADER fileHeader;
    BITMAPINFOHEADER infoHeader;

    fread(&fileHeader, sizeof(fileHeader), 1, fp);
    fread(&infoHeader, sizeof(infoHeader), 1, fp);

    if (fileHeader.bfType != 0x4D42) {
        printf("BMPでない\n");
        return 1;
    }

    if (infoHeader.biBitCount != 24) {
        printf("24bit限定\n");
        return 1;
    }

    Image img;

    img.width  = infoHeader.biWidth;
    img.height = infoHeader.biHeight;
    img.rowSize = ((img.width * 3 + 3) / 4) * 4;

    img.data = malloc(img.rowSize * img.height);
    if (!img.data) {
        printf("メモリ確保失敗\n");
        return 1;
    }

    fseek(fp, fileHeader.bfOffBits, SEEK_SET);
    fread(img.data, 1, img.rowSize * img.height, fp);
    fclose(fp);

    Image dst; // 出力用バッファ
    dst.width = img.width;
    dst.height = img.height;
    dst.rowSize = img.rowSize;
    dst.data = malloc(dst.rowSize * dst.height);

/********************************************
/***     ここに個別の実験用コードを配置      ***
/********************************************

    // =========================
    // 書き出し
    // =========================

    FILE *out = fopen("output.bmp", "wb");
    if (!out) {
        printf("output.bmp 開けない\n");
        free(img.data);
        return 1;
    }

    fwrite(&fileHeader, sizeof(fileHeader), 1, out);
    fwrite(&infoHeader, sizeof(infoHeader), 1, out);
    fwrite(dst.data, 1, dst.rowSize * dst.height, out);

    fclose(out);
    free(img.data);

    printf("output.bmp 生成完了\n");

    return 0;
}
```

# 拡大と平行移動

- x,y両方向にそれぞれ2倍する。
- x方向に50ピクセル移動する。
- y方向に30ピクセル移動する。

<img width="642" height="328" alt="image" src="https://github.com/user-attachments/assets/0ea78e31-4c17-4a25-858f-cdf61beef3c3" />

```c
    double sx = 2.0; // 拡大率
    double sy = 2.0;
    int tx = 50;     // 平行移動
    int ty = 30;

    for (int y = 0; y < img.height; y++) {
        for (int x = 0; x < img.width; x++) {

            unsigned char *p = get_pixel(&img, x, y);
            if (!p) continue;

            int nx = (int)(x * sx) + tx;
            int ny = (int)(y * sy) + ty;

            unsigned char *dp = get_pixel(&dst, nx, ny);
            if (!dp) continue;

            dp[0] = p[0];
            dp[1] = p[1];
            dp[2] = p[2];
        }
    }
```

# 回転

<img width="634" height="321" alt="image" src="https://github.com/user-attachments/assets/c46e1545-6bb9-4938-aaba-9d680f2ca5b0" />

```c
    double theta = 30.0 * M_PI / 180.0; // 30度
    double cos_t = cos(theta);
    double sin_t = sin(theta);
	
    for (int y = 0; y < img.height; y++) {
        for (int x = 0; x < img.width; x++) {

            unsigned char *p = get_pixel(&img, x, y);
            if (!p) continue;

            // 回転
            int nx = (int)(x * cos_t - y * sin_t);
            int ny = (int)(x * sin_t + y * cos_t);

            unsigned char *dp = get_pixel(&dst, nx, ny);
            if (!dp) continue;

            dp[0] = p[0];
            dp[1] = p[1];
            dp[2] = p[2];
        }
    }
```

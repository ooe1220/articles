```my_bmp.c
#include <stdio.h>
#include <stdlib.h>
#include "my_bmp.h"

#pragma pack(1)
typedef struct {
    unsigned short bfType;
    unsigned int   bfSize;
    unsigned short bfReserved1;
    unsigned short bfReserved2;
    unsigned int   bfOffBits;
} BITMAPFILEHEADER;

typedef struct {
    unsigned int   biSize;
    int            biWidth;
    int            biHeight;
    unsigned short biPlanes;
    unsigned short biBitCount;
    unsigned int   biCompression;
    unsigned int   biSizeImage;
    int            biXPelsPerMeter;
    int            biYPelsPerMeter;
    unsigned int   biClrUsed;
    unsigned int   biClrImportant;
} BITMAPINFOHEADER;
#pragma pack()

/**
 * @brief BMPファイル（24bit）を読み込み、Image構造体に格納する
 *
 * 指定したBMPファイルを読み込み、画像サイズ・行サイズを設定し、
 * 画素データをヒープに確保してimg->dataへ格納する。
 *
 * 対応形式は「24bit 非圧縮BMP」のみ。
 *
 * @param filename 入力ファイル名（例: "input.bmp"）
 * @param img      出力先Image構造体
 *                 - width, height, rowSize が設定される
 *                 - data は関数内でmallocされる
 *
 * @return int
 *         0   : 成功
 *        -1   : ファイル開くのに失敗
 *        -2   : BMP形式ではない（bfType不正）
 *        -3   : 24bit以外の形式
 *        -4   : メモリ確保失敗
 *
 * @note
 * - 読み込まれるデータはBGR順で格納される
 * - 行はBMP仕様に従い「下から上」の順で格納される
 * - rowSizeは4バイトアラインされた値になる
 *
 * @warning
 * - imgがNULLの場合は未定義動作
 * - 読み込み成功後は必ずfree_image()で解放すること
 * - 圧縮BMPやパレット形式（8bitなど）には未対応
 */
int load_bmp(const char *filename, Image *img) {
    FILE *fp = fopen(filename, "rb");
    if (!fp) return -1;

    BITMAPFILEHEADER fileHeader;
    BITMAPINFOHEADER infoHeader;

    fread(&fileHeader, sizeof(fileHeader), 1, fp);
    fread(&infoHeader, sizeof(infoHeader), 1, fp);

    if (fileHeader.bfType != 0x4D42) { // BMP形式でない
        fclose(fp);
        return -2;
    }

    if (infoHeader.biBitCount != 24) { // 24bitでない
        fclose(fp);
        return -3;
    }

    img->width  = infoHeader.biWidth;
    img->height = infoHeader.biHeight;
    img->rowSize = ((img->width * 3 + 3) / 4) * 4; // BITMAPの仕様により4の倍数(切り上げ)に合わす

    img->data = malloc(img->rowSize * img->height);
    if (!img->data) {
        fclose(fp);
        return -4;
    }

    fseek(fp, fileHeader.bfOffBits, SEEK_SET);//fpをピクセルのデータまで進める
    fread(img->data, 1, img->rowSize * img->height, fp); // ピクセルデータをdataへ格納

    fclose(fp);
    return 0;
}

/**
 * @brief BMPファイル（24bit）として画像を書き出す
 *
 * 指定したImage構造体のデータを、24bit非圧縮BMP形式でファイルに保存する。
 * ヘッダは関数内で自動生成される。
 *
 * @param filename 出力ファイル名（例: "output.bmp"）
 * @param img      書き出す画像データ
 *                 - width   : 画像の幅（ピクセル）
 *                 - height  : 画像の高さ（ピクセル）
 *                 - rowSize : 1行のバイト数（4バイト揃え済み）
 *                 - data    : 画素データ（BGR順、下から上の行順）
 *
 * @return int
 *         0  : 成功
 *        -1  : 失敗
 *
 * @note
 * - 24bit BMP（biBitCount = 24）のみ対応
 * - 圧縮形式（RLEなど）には未対応
 * - img->data は事前に確保・初期化されている必要がある
 * - ファイルは上書きされる
 *
 * @warning
 * - imgがNULL、またはdataがNULLの場合の動作は未定義
 * - rowSizeが不正だと出力ファイルは壊れる
 */
int save_bmp(const char *filename, Image *img) {
    FILE *fp = fopen(filename, "wb");
    if (!fp) return -1;

    BITMAPFILEHEADER fileHeader = {0};
    BITMAPINFOHEADER infoHeader = {0};

    fileHeader.bfType = 0x4D42;
    fileHeader.bfOffBits = sizeof(fileHeader) + sizeof(infoHeader);
    fileHeader.bfSize = fileHeader.bfOffBits + img->rowSize * img->height;

    infoHeader.biSize = sizeof(infoHeader);
    infoHeader.biWidth = img->width;
    infoHeader.biHeight = img->height;
    infoHeader.biPlanes = 1;
    infoHeader.biBitCount = 24;
    infoHeader.biSizeImage = img->rowSize * img->height;

    fwrite(&fileHeader, sizeof(fileHeader), 1, fp);
    fwrite(&infoHeader, sizeof(infoHeader), 1, fp);
    fwrite(img->data, 1, img->rowSize * img->height, fp);

    fclose(fp);
    return 0;
}

/**
 * @brief 指定座標のピクセルへのポインタを取得する
 *
 * 画像上の(x, y)座標に対応するピクセルデータ（3バイト: BGR）への
 * ポインタを返す。座標系は左上原点（yは下方向に増加）として扱う。
 *
 * 内部的にはBMP形式（下から上へ格納）に対応するため、
 * y座標を反転してアドレスを計算している。
 *
 * @param img 対象のImage構造体
 * @param x   x座標（0 <= x < width）
 * @param y   y座標（0 <= y < height）
 *
 * @return unsigned char*
 *         - 有効な場合：ピクセル先頭アドレス（p[0]=B, p[1]=G, p[2]=R）
 *         - 範囲外の場合：NULL
 *
 * @note
 * - 返されるポインタは3バイトの連続領域を指す
 * - 書き込み可能（set_rgbなどで使用可能）
 *
 * @warning
 * - imgがNULLの場合は未定義動作
 * - rowSizeが不正な場合は誤った位置を指す
 */
unsigned char* get_pixel(Image *img, int x, int y) {
    if (x < 0 || x >= img->width || y < 0 || y >= img->height) return NULL;
    return img->data + (img->height - 1 - y) * img->rowSize + x * 3;
}

/**
 * @brief ピクセルにRGB値を設定する（24bit BMP用）
 *
 * 指定したピクセル領域に対して、RGB値を設定する。
 * BMP形式ではBGR順で格納されるため、内部的には b, g, r の順で書き込む。
 *
 * @param p 書き込み先ピクセルへのポインタ
 * @param r 赤成分（0〜255）
 * @param g 緑成分（0〜255）
 * @param b 青成分（0〜255）
 *
 * @note
 * - pは通常、get_pixel() から取得したポインタを想定
 * - BMPはBGR順のため、引数の順番（RGB）とメモリ配置が異なる点に注意
 *
 * @warning
 * - pがNULLの場合は未定義動作（呼び出し側で確認すること）
 * - r, g, bが0〜255の範囲外でも処理は行わない
 */
void set_rgb(unsigned char *p, int r, int g, int b) {
    p[0] = b;
    p[1] = g;
    p[2] = r;
}

/**
 * @brief Image構造体が保持するメモリを解放する
 *
 * img->data に確保されている画素データを解放し、
 * ポインタをNULLに設定することで二重解放を防ぐ。
 *
 * @param img 解放対象のImage構造体
 *
 * @note
 * - img自体（構造体本体）は解放しない
 * - dataがNULLの場合は何も行わない
 *
 * @warning
 * - imgがNULLの場合は未定義動作
 * - この関数呼び出し後、dataを参照してはいけない
 */
void free_image(Image *img) {
    if (img->data) {
        free(img->data);
        img->data = NULL;
    }
}
```

```main.c
//gcc main.c my_bmp.c -o main
#include "my_bmp.h"
#include <stdlib.h>
#include <math.h>

int main() {
    Image img, dst;

    load_bmp("input.bmp", &img);
	
    double theta = 0.1;
    double cos_t = cos(theta);
    double sin_t = sin(theta);

    dst = img;
    dst.data = malloc(img.rowSize * img.height);
	
    // 黒で初期化
    for (int i = 0; i < img.rowSize * img.height; i++) {
        dst.data[i] = 0;
    }

    // 実験
    for (int y = 0; y < img.height; y++) {
        for (int x = 0; x < img.width; x++) {
            
            // 回転行列で回転させる
            int dx = (int)(x * cos_t - y * sin_t);
            int dy = (int)(x * sin_t + y * cos_t);

            unsigned char *sp = get_pixel(&img, x, y);
            unsigned char *dp = get_pixel(&dst, dx, dy);

            if (sp && dp) {
                dp[0] = sp[0];
                dp[1] = sp[1];
                dp[2] = sp[2];
            }
        }
    }

    save_bmp("output.bmp", &dst);

    free_image(&img);
    free_image(&dst);
}
```
